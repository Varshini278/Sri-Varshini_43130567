package nov21;
abstract class Animals{
	 
	  public abstract void animalSound();
	 
	  public void sleep() {
	    System.out.println("Zzz");
	  }
	}

	class Pig extends Animals{
	  public void animalSound() {
	    System.out.println("The pig says: wee wee");
	  }
	}

	class Abstract {
	  public static void main(String[] args) {
	    Pig myPig = new Pig(); 
	    myPig.animalSound();
	    myPig.sleep();
	  }
	}
